library(magrittr)

set <- .Primitive(“[<-”) get <- .Primitive(“[[”)

zz <- list(a = 1, b = 2, c = 3) zz %>% set(“d”, 4) zz %>% get(“b”)

select <- function(x, cols) { sel <- .Primitive(“[”) sel(x, 1:nrow(x), cols, drop = FALSE) } mtcars %>% select(“mpg”) mtcars %>% select(c(“mpg”, “cyl”))

pull <- function(x, cols) { sel <- .Primitive(“[”) sel(x, 1:nrow(x), cols) } mtcars %>% pull(“mpg”)

filter <- function(x, i) { filt <- .Primitive(“[”) filt(x, i, colnames(x)) } mtcars %>% filter(mtcars\(cyl == 4) mtcars %>% filter(mtcars\)cyl == 4 & mtcars$am == 0)

mutate <- function(x, name, new_col) { mut <- .Primitive(“[<-”) mut(x, 1:nrow(x), name, new_col) } mtcars %>% mutate(“mpg_squared”, mtcars\(mpg ^ 2) mtcars %>% mutate(c("mpg_squared", "cyl_div_2"), c(mtcars\)mpg ^ 2, mtcars$cyl / 2))
